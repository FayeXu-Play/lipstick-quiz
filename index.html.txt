<script>
    let scores = { type: '', pref: '' };

    function startQuiz() {
        document.getElementById('intro').style.display = 'none';
        document.getElementById('q1').classList.add('active');
    }

    function nextStep(type) {
        scores.type = type;
        document.getElementById('q1').classList.remove('active');
        document.getElementById('q2').classList.add('active');
    }

    function collectData(pref) {
        scores.pref = pref;
        showResult();
    }

    function showResult() {
        document.getElementById('q2').classList.remove('active');
        const resDiv = document.getElementById('result');
        const name = document.getElementById('lipstick-name');
        const desc = document.getElementById('lipstick-desc');
        resDiv.style.display = 'block';

        if (scores.type === 'matte') {
            if (scores.pref === 'A') {
                name.innerText = "931 Nude Apricot | The Intellectual Philosopher";
                desc.innerText = "Rational yet gentle. Like a scholar pondering in a Left Bank café, your 'Effortless Wisdom' is your power.";
            } else {
                name.innerText = "937 Palpitating Rose | The Ethereal Artist";
                desc.innerText = "Independent and composed. This cool-toned rose complements your sharp, elegant, and untouchable vibe.";
            }
        } else {
            if (scores.pref === 'B') {
                name.innerText = "703 Gentle Cinnamon | Master of Effortless Chic";
                desc.innerText = "The ultimate 'I woke up like this'. You don't do extra—you do better. Naturally radiant and relaxed.";
            } else if (scores.pref === 'C') {
                name.innerText = "721 Posh Guava | The Luminous Protagonist";
                desc.innerText = "Your soul sparkles with energy. You don't just attend the party; you ARE the atmosphere. Radiant and refined.";
            } else {
                name.innerText = "712 Lucky Koi | High-Energy Lucky Charm";
                desc.innerText = "The sunshine student! Even during finals week, you carry the petit bonheur. Fresh, energized, and ready.";
            }
        }
    }
</script>
