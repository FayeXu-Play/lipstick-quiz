<script>
    // Initial scores for each lipstick
    let scores = { "931": 0, "937": 0, "703": 0, "712": 0, "721": 0 };
    let group = ""; // matte or moist

    function showStep(n) {
        document.querySelectorAll('.question, #intro').forEach(el => el.style.display = 'none');
        if(n <= 4) document.getElementById('step' + n).classList.add('active');
    }

    function vote(key, points) {
        if(key === 'matte' || key === 'moist') {
            group = key;
        } else {
            scores[key] += points;
        }
    }

    function finish() {
        document.querySelectorAll('.question').forEach(el => el.classList.remove('active'));
        const resultDiv = document.getElementById('result');
        const nameEl = document.getElementById('lipstick-name');
        const descEl = document.getElementById('lipstick-desc');
        
        // Final logic: Filter by group first, then find highest score
        let pool = (group === 'matte') ? ["931", "937"] : ["703", "712", "721"];
        let winner = pool.reduce((a, b) => scores[a] > scores[b] ? a : b);

        const data = {
            "931": ["931 Nude Apricot | The Intellectual Philosopher", "A warm apricot brown. You balance logic with a gentle soul. This shade is your 'Effortless Wisdom'—perfect for deep conversations in a cozy café."],
            "937": ["937 Palpitating Rose | The Ethereal Artist", "A cool-toned pink rose. You are independent, sharp, and undeniably chic. This is your power color for conquering the city with an elegant edge."],
            "703": ["703 Gentle Cinnamon | Master of Effortless Chic", "A natural, moisturizing glow. You embody the true French '松弛感' (relaxation). You look stunning without even trying."],
            "712": ["712 Lucky Koi | The High-Energy Lucky Charm", "Fresh and vibrant. You are the 'Morning Star' energy. This shade brings you a boost of luck and a bright, healthy complexion for those early lectures."],
            "721": ["721 Posh Guava | The Luminous Protagonist", "Radiant and high-class. You have a natural gift for standing out. This guava shade highlights your elegant, 'C-position' charisma."]
        };

        nameEl.innerText = data[winner][0];
        descEl.innerText = data[winner][1];
        resultDiv.style.display = 'block';
    }
</script>
