<script>
    let tags = { "Cool": 0, "Gentle": 0, "Relax": 0, "Power": 0, "Posh": 0 };
    let group = ""; // matte or moist

    function showStep(n) {
        document.querySelectorAll('.question, #intro').forEach(el => el.style.display = 'none');
        if(n <= 4) {
            document.getElementById('step' + n).classList.add('active');
            document.getElementById('step' + n).style.display = 'block';
        }
    }

    function setGroup(val) { group = val; }
    function addTag(tag, pts) { tags[tag] += pts; }

    function finish() {
        document.querySelectorAll('.question').forEach(el => el.style.display = 'none');
        const nameEl = document.getElementById('lipstick-name');
        const descEl = document.getElementById('lipstick-desc');
        
        let winner = "";
        
        if (group === 'matte') {
            // PK Cool vs Gentle
            winner = (tags["Cool"] + tags["Posh"] >= tags["Gentle"] + tags["Relax"]) ? "937" : "931";
        } else {
            // PK Power vs Relax vs Posh
            let pool = { "712": tags["Power"], "703": tags["Relax"], "721": tags["Posh"] };
            winner = Object.keys(pool).reduce((a, b) => pool[a] > pool[b] ? a : b);
        }

        const content = {
            "931": ["931 Nude Apricot | The Intellectual Philosopher", "A warm apricot brown. You balance logic with a gentle soul. This shade is your 'Effortless Wisdom'—perfect for deep conversations in a cozy Left Bank café."],
            "937": ["937 Palpitating Rose | The Ethereal Artist", "A cool-toned pink rose. You are independent, sharp, and undeniably chic. This is your power color for conquering the city with an elegant edge."],
            "703": ["703 Gentle Cinnamon | Master of Effortless Chic", "A natural, moisturizing glow. You embody the true French '松弛感' (relaxation). You look stunning without even trying."],
            "712": ["712 Lucky Koi | The High-Energy Lucky Charm", "Fresh and vibrant. The 'Morning Star' energy. This shade brings a boost of luck and a bright, healthy complexion for your busiest days."],
            "721": ["721 Posh Guava | The Luminous Protagonist", "Radiant and high-class. You have a natural gift for standing out. This guava shade highlights your elegant, 'C-position' charisma."]
        };

        nameEl.innerText = content[winner][0];
        descEl.innerText = content[winner][1];
        document.getElementById('result').style.display = 'block';
    }
</script>
